Mysql-InnoDBзҙўеј•д№Ӣзҙўеј•дҪҝз”Ёе’ҢдјҳеҢ–еҺҹеҲҷ(дәҢ)

е’•еҷңгҖӮе’•еҷң

mysql

еҸ‘еёғдәҺпјҡJul 18, 2019

жҸҗеҮәй—®йўҳ

  1. д»Җд№ҲжҳҜеӣһиЎЁпјҹ
  2. зҙўеј•жҳҜдёҚжҳҜеҸҜд»Ҙж— йҷҗе»әпјҹ
  3. д»Җд№ҲжҳҜзҙўеј•зҡ„жңҖе·ҰеҢ№й…ҚеҺҹеҲҷпјҹ
  4. д»Җд№ҲжҳҜиҰҶзӣ–зҙўеј•пјҹ
  5. еҰӮдҪ•е»әз«Ӣй«ҳж•Ҳзҡ„зҙўеј•пјҹ

зҙўеј•зҡ„д»Јд»·

йҖҡиҝҮдёҠж–ҮгҖҠMysql-InnoDBзҙўеј•д№Ӣзҙўеј•з»“жһ„иҜҰи§Ј(дёҖ)гҖӢжҲ‘д»¬дәҶи§ЈеҲ°дәҶB+ж ‘зҙўеј•зҡ„еҺҹзҗҶгҖӮInnoDBеӯҳеӮЁеј•ж“ҺдјҡвҫғеҠЁдёәдё»й”®пјҲеҰӮжһңжІЎжңүе®ғдјҡвҫғеҠЁеё®жҲ‘д»¬ж·»еҠ пјүе»әвҪҙиҒҡз°Үзҙўеј•пјҢиҒҡз°Үзҙўеј•зҡ„еҸ¶вјҰиҠӮзӮ№еҢ…еҗ«е®Ңж•ҙзҡ„вҪӨжҲ·и®°еҪ•гҖӮиҖҢдё”жҜҸж¬Ўе»әз«ӢдёҖж¬Ўзҙўеј•йғҪйңҖиҰҒдёәе…¶е»әз«ӢдёҖйў—B+ж ‘гҖӮ

з©әй—ҙдёҠзҡ„д»Јд»·

жҜҸе»әвҪҙвјҖдёӘзҙўеј•йғҪиҰҒдёәе®ғе»әвҪҙвјҖжЈөB+ж ‘пјҢжҜҸвјҖжЈөB+ж ‘зҡ„жҜҸвјҖдёӘиҠӮзӮ№йғҪжҳҜвјҖдёӘж•°жҚ®в»ҡпјҢвјҖдёӘв»ҡй»ҳи®ӨдјҡеҚ вҪӨ16KBзҡ„еӯҳеӮЁз©әй—ҙпјҢвјҖжЈөеҫҲвјӨзҡ„B+ж ‘з”ұи®ёеӨҡж•°жҚ®в»ҡз»„жҲҗпјҢйӮЈеҸҜжҳҜеҫҲвјӨзҡ„вјҖвҪҡеӯҳеӮЁз©әй—ҙгҖӮ

ж—¶й—ҙдёҠзҡ„д»Јд»·

жҜҸж¬ЎеҜ№иЎЁдёӯзҡ„ж•°жҚ®иҝӣвҫҸеўһгҖҒеҲ гҖҒж”№ж“ҚдҪңж—¶пјҢйғҪйңҖиҰҒеҺ»дҝ®ж”№еҗ„дёӘB+ж ‘зҙўеј•гҖӮжҲ‘д»¬жё…жҘҡB+ж ‘жҜҸеұӮиҠӮзӮ№йғҪжҳҜжҢүз…§зҙўеј•еҲ—зҡ„еҖјд»Һе°ҸеҲ°еӨ§йЎәеәҸжҺ’еәҸзҡ„пјҢеҰӮжһңеҒҡдәҶеўһеҲ ж”№еҲҷдјҡеҜ№жҺ’еәҸйҖ жҲҗз ҙеқҸпјҢжүҖд»ҘеӯҳеӮЁеј•ж“ҺйңҖиҰҒйўқеӨ–зҡ„ж—¶й—ҙиҝӣвҫҸвјҖдәӣи®°еҪ•з§»дҪҚпјҢв»ҡвҫҜеҲҶиЈӮгҖҒв»ҡвҫҜеӣһж”¶е•Ҙзҡ„ж“ҚдҪңжқҘз»ҙжҠӨеҘҪиҠӮзӮ№е’Ңи®°еҪ•зҡ„жҺ’еәҸпјҢжүҖд»ҘеңЁж—¶й—ҙдёҠйңҖиҰҒеҜ№зҙўеј•иҝӣиЎҢз»ҙжҠӨгҖӮ

жүҖд»ҘпјҢдёҖдёӘиЎЁдёҠзҙўеј•е»әзҡ„и¶ҠеӨҡпјҢе°ұдјҡеҚ вҪӨи¶ҠеӨҡзҡ„еӯҳеӮЁз©әй—ҙпјҢеңЁеўһеҲ ж”№и®°еҪ•зҡ„ж—¶еҖҷжҖ§иғҪе°ұи¶Ҡе·®гҖӮ

зҙўеј•зҡ„дҪҝз”Ё

жҲ‘д»¬е…Ҳе»әз«ӢдёҖдёӘиЎЁ

CREATE TABLE person_info (
        id INT NOT NULL auto_increment,
        NAME VARCHAR (100) NOT NULL,
        birthday DATE NOT NULL,
        phone_number CHAR (11) NOT NULL,
        country VARCHAR (100) NOT NULL,
        PRIMARY KEY (id),
        KEY idx_name_birthday_phone_number (NAME, birthday, phone_number)
);

ж №жҚ®дёҠйқўзҡ„SQLпјҢжҲ‘д»¬зҹҘйҒ“иЎЁperson_infoзҡ„зҙўеј•жғ…еҶө

  • дё»й”®жҳҜidеҲ—пјҢInnoDBеӯҳеӮЁеј•ж“Һдёәдё»й”®и®ҫзҪ®дәҶдёҖдёӘиҒҡз°Үзҙўеј•пјҢеӣ дёәжҳҜиҒҡз°Үзҙўеј•зҙўеј•еҜ№еә”зҡ„B+ж ‘зҡ„еҸ¶еӯҗиҠӮзӮ№еӯҳеӮЁзҡ„з”ЁжҲ·и®°еҪ•еҢ…еҗ«жүҖжңүеҲ—
  • е®ҡд№үдәҶдёҖдёӘиҒ”еҗҲзҙўеј•idx_name_birthday_phone_number,з”ұеҲ—nameпјҢbirthdayпјҢphone_numberз»„жҲҗпјҢиҝҷдёӘиҒ”еҗҲзҙўеј•зҡ„B+ж ‘зҡ„еҸ¶еӯҗиҠӮзӮ№еӯҳеӮЁзҡ„з”ЁжҲ·и®°еҪ•еҢ…еҗ«nameпјҢbirthdayпјҢphone_numbeе’Ңдё»й”®idпјҢдҪҶдёҚеҢ…еҗ«countryгҖӮ

иҝҷйҮҢйңҖиҰҒжіЁж„Ҹзҡ„жҳҜиҒ”еҗҲзҙўеј•B+ж ‘зҡ„е»әз«ӢжҳҜжҢүз…§е…Ҳname,еҶҚbirthday,жңҖеҗҺphone_numberзҡ„йЎәеәҸпјҢзҙўеј•B+ж ‘зҡ„йЎәеәҸжҳҜжҢүз…§

  • е…ҲйҖҡиҝҮnameиҝӣиЎҢжҺ’еәҸгҖӮ
  • еҰӮжһңnameзӣёеҗҢпјҢеҶҚжҢүз…§birthdayжҺ’еәҸгҖӮ
  • еҰӮжһңbirthdayд№ҹзӣёеҗҢпјҢеҲҷжҢүз…§phone_numberжҺ’еәҸгҖӮ

йҖҡиҝҮдё»й”®IDжҹҘжүҫж•°жҚ®

SELECT
    *
FROM
    person_info
WHERE
    ID = 10;

иҝҷз§Қж–№ејҸдёӢпјҢз”ұдәҺIDжҳҜдё»й”®пјҢзҙўеј•йҖҡиҝҮиҒҡз°Үзҙўеј•жүҖеңЁзҡ„B+ж ‘ж №жҚ®дәҢеҲҶжі•жҹҘжүҫеҲ°еҜ№еә”зҡ„ж•°жҚ®гҖӮ

е…ЁеҖјеҢ№й…Қ

SELECT
    *
FROM
    person_info
WHERE
    NAME = 'Ashburn'
AND birthday = '1990-09-27'
AND phone_number = '15123983239';

дёҠйқўSQLпјҢз”ұдәҺжҹҘиҜўжқЎд»¶еҢ…еҗ«еңЁдәҶidx_name_birthday_phone_numberзҙўеј•еҶ…пјҢжүҖд»ҘжҹҘиҜўиҝҮзЁӢеҰӮдёӢ

  1. еӣ дёәB+ж ‘зҡ„ж•°жҚ®в»ҡе’Ңи®°еҪ•е…ҲжҳҜжҢүз…§nameеҲ—зҡ„еҖјиҝӣвҫҸжҺ’еәҸзҡ„пјҢжүҖд»Ҙе…ҲеҸҜд»ҘеҫҲеҝ«е®ҡдҪҚnameеҲ—зҡ„еҖјжҳҜAshburnзҡ„и®°еҪ•дҪҚзҪ®
  2. еңЁnameеҲ—зӣёеҗҢзҡ„и®°еҪ•вҫҘвјңжҳҜжҢүз…§birthdayеҲ—зҡ„еҖјиҝӣвҫҸжҺ’еәҸзҡ„пјҢжүҖд»ҘеңЁnameеҲ—зҡ„еҖјжҳҜAshburnзҡ„и®°еҪ•вҫҘвјңеҸҜд»Ҙеҝ«йҖҹе®ҡдҪҚbirthdayеҲ—зҡ„еҖјжҳҜвҖҷ1990-09-27вҖҷзҡ„и®°еҪ•
  3. еҰӮжһңеҫҲдёҚе№ёпјҢnameе’ҢbirthdayеҲ—зҡ„еҖјйғҪжҳҜзӣёеҗҢзҡ„пјҢйӮЈи®°еҪ•жҳҜжҢүз…§phone_numberеҲ—зҡ„еҖјжҺ’еәҸзҡ„пјҢжүҖд»ҘиҒ”еҗҲзҙўеј•дёӯзҡ„дёүдёӘеҲ—йғҪеҸҜиғҪиў«вҪӨеҲ°

иҝҷз§Қжғ…еҶөдёӢжҲ‘д»¬йҖҡиҝҮдёүдёӘеҲ—зҡ„е…ЁеҖјеҢ№й…ҚжҹҘжүҫеҲ°зҡ„еҜ№еә”зҡ„и®°еҪ•пјҢи®°еҪ•дёҠеҢ…еҗ«дәҶid,name,birthday,phone_numberгҖӮдҪҶжҳҜжҲ‘д»¬жҹҘиҜўзҡ„жҳҜ*,еңЁidx_name_birthday_phone_numberзҙўеј•B+ж ‘дёҠе№¶жІЎжңүеӯҳеӮЁcountryзҡ„ж•°жҚ®пјҢйӮЈиҝҷж—¶еҖҷеҰӮдҪ•жҹҘжүҫcountryпјҹ

еӣһиЎЁ

з”ұдәҺcountryеҸӘи®°еҪ•еңЁиҒҡз°Үзҙўеј•B+ж ‘дёҠпјҢиҖҢжҲ‘д»¬йҖҡиҝҮиҒ”еҗҲзҙўеј•е·Із»ҸжҹҘеҲ°ж•°жҚ®еҜ№еә”зҡ„IDпјҢжүҖд»Ҙиҝҷж—¶еҖҷйҖҡиҝҮIDз»§з»ӯеҺ»иҒҡз°Үзҙўеј•дёҠжҹҘжүҫдёҖйҒҚпјҢжңҖз»ҲиҺ·еҸ–еҲ°дәҶжңҖеҗҺзҡ„ж•°жҚ®гҖӮ

иҝҷз§ҚйңҖиҰҒжҠҠд»ҺдёҠвјҖжӯҘдёӯиҺ·еҸ–еҲ°зҡ„жҜҸвјҖжқЎи®°еҪ•зҡ„idеӯ—ж®өйғҪеҲ°иҒҡз°Үзҙўеј•еҜ№еә”зҡ„B+ж ‘дёӯжүҫеҲ°е®Ңж•ҙзҡ„вҪӨжҲ·и®°еҪ•пјҢд№ҹе°ұжҳҜжҲ‘д»¬йҖҡеёёжүҖиҜҙзҡ„еӣһиЎЁ

еӣһиЎЁзҡ„д»Јд»·

йңҖиҰҒеӣһиЎЁзҡ„и®°еҪ•и¶ҠеӨҡпјҢдҪҝвҪӨвјҶзә§зҙўеј•зҡ„жҖ§иғҪе°ұи¶ҠдҪҺпјҢз”ҡвҫ„и®©жҹҗдәӣжҹҘиҜўе®Ғж„ҝдҪҝвҪӨе…ЁиЎЁжү«жҸҸд№ҹдёҚдҪҝвҪӨвјҶзә§зҙўеј•гҖӮеҒҮи®ҫжңүдёҖз§Қжғ…еҶөпјҢname = вҖҳAshburnвҖҷ AND birthday = вҖҳ1990-09-27вҖҷ AND phone_number = вҖҳ15123983239вҖҷ зҡ„ж•°жҚ®еҚ дәҶжүҖжңүж•°жҚ®зҡ„90%д»ҘдёҠпјҢйӮЈжӯӨж—¶йҖҡиҝҮиҒ”еҗҲзҙўеј•е…ҲжҹҘдёҖйҒҚиҺ·еҸ–еҲ°еҜ№еә”зҡ„дё»й”®IDеҶҚеҺ»иҒҡз°Үзҙўеј•жҹҘдёҖйҒҚзҡ„ж–№ејҸпјҲдәҢзә§зҙўеј•+еӣһиЎЁпјүе°ұжҜ”е…ЁиЎЁжү«жҸҸж•ҲзҺҮе·®гҖӮ

жүҖд»ҘеңЁдҪҝз”ЁдёҠжҲ‘д»¬еә”иҜҘе°ҪйҮҸйҒҝе…ҚеӣһиЎЁдә§з”ҹзҡ„й—®йўҳ

  • е»әи®®зӮ№1 пјҡдҫӢеҰӮеҸҜд»ҘеҲӨж–ӯдёҡеҠЎдёҠжҳҜеҗҰйңҖиҰҒиҺ·еҸ–жүҖжңүеҲ—зҡ„ж•°жҚ®пјҢеҰӮжһңдёҡеҠЎдёҠеҸӘйңҖиҰҒid,name,birthday,phone_numberеҲ—зҡ„еҖјиҖҢдёҚйңҖиҰҒcountryзҡ„еҖјпјҢиҝҷж—¶еҖҷеҸҜд»ҘдёҚдҪҝз”Ёselect * ,иҝҷж ·еҸҜд»ҘйҳІжӯўеӣһиЎЁ
  • е»әи®®зӮ№2 пјҡе°ҪйҮҸйҒҝе…ҚеӣһиЎЁзҡ„ж¬Ўж•°пјҢиҝҷз§ҚеҸҜд»Ҙд»Һзҙўеј•зҡ„йҖүеҸ–ж–№йқўиҖғиҷ‘пјҢдҫӢеҰӮеҜ№дәҺйҮҚеӨҚеҖјеӨ§зҡ„еҲ—дёҚе»әи®®и®ҫзҪ®дёәзҙўеј•гҖӮ

иҰҶзӣ–зҙўеј•

дёәдәҶеҪ»еә•е‘ҠеҲ«еӣһиЎЁж“ҚдҪңеёҰжқҘзҡ„жҖ§иғҪжҚҹиҖ—пјҢжҲ‘д»¬е»әи®®пјҡжңҖеҘҪеңЁжҹҘиҜўеҲ—иЎЁвҫҘеҸӘеҢ…еҗ«зҙўеј•еҲ—

SELECT
    ID,NAME,birthday,phone_number
FROM
    person_info
WHERE
    NAME = 'Ashburn'
AND birthday = '1990-09-27'
AND phone_number = '15123983239';

жҲ‘д»¬жҠҠиҝҷз§ҚеҸӘйңҖиҰҒвҪӨеҲ°зҙўеј•зҡ„жҹҘиҜўвҪ…ејҸз§°дёәзҙўеј•иҰҶзӣ–гҖӮжҺ’еәҸж“ҚдҪңд№ҹдјҳе…ҲдҪҝвҪӨиҰҶзӣ–зҙўеј•

жңҖе·ҰеҢ№й…ҚеҺҹеҲҷ

дёӢйқўдёӨдёӘSQLиҜӯеҸҘпјҢжҹҘиҜўжқЎд»¶зӣёеҗҢпјҢеҸӘжҳҜйЎәеәҸдёҚеҗҢпјҢжҳҜеҗҰжҖ§иғҪзӣёеҗҢпјҹ

-- SQL 1
SELECT
    *
FROM
    person_info
WHERE
    NAME = 'Ashburn'
AND birthday = '1990-09-27'
AND phone_number = '15123983239';
-- SQL 2
SELECT
    *
FROM
    person_info
WHERE
    birthday = '1990-09-27'
AND NAME = 'Ashburn'
AND phone_number = '15123983239';

зӯ”жЎҲжҳҜдёҚзӣёеҗҢпјҢйҖҡиҝҮзҙўеј•зҡ„еҲҶжһҗпјҢжҲ‘д»¬зҹҘйҒ“idx_name_birthday_phone_numberзҡ„жҺ’еәҸж–№ејҸжҳҜname -> birthday - > phone_numberгҖӮдҪҶжҳҜSQL2жҳҜе…Ҳbirthday->NAME->phone_numberгҖӮиҝҷж ·еҲҷжІЎеҠһжі•е…ҲйҖҡиҝҮnameеҺ»жҹҘжүҫзҙўеј•пјҢжүҖд»ҘSQL2дҪҝз”ЁдёҚеҲ°зҙўеј•е°ҶдјҡйҮҮз”Ёе…ЁиЎЁжү«жҸҸгҖӮ

жүҖд»ҘпјҢеҰӮжһңжҲ‘д»¬жғідҪҝвҪӨиҒ”еҗҲзҙўеј•дёӯе°ҪеҸҜиғҪеӨҡзҡ„еҲ—пјҢжҗңзҙўжқЎд»¶дёӯзҡ„еҗ„дёӘеҲ—еҝ…йЎ»жҳҜиҒ”еҗҲзҙўеј•дёӯд»ҺжңҖе·Ұиҫ№иҝһз»ӯзҡ„еҲ—

з”ЁдәҺжҺ’еәҸ

  1. ASC,DESCж··з”ЁпјҢеҜ№дәҺдҪҝвҪӨиҒ”еҗҲзҙўеј•иҝӣвҫҸжҺ’еәҸзҡ„еңәжҷҜпјҢжҲ‘д»¬иҰҒжұӮеҗ„дёӘжҺ’еәҸеҲ—зҡ„жҺ’еәҸйЎәеәҸжҳҜвјҖиҮҙзҡ„пјҢд№ҹе°ұжҳҜиҰҒд№Ҳеҗ„дёӘеҲ—йғҪжҳҜASCи§„еҲҷжҺ’еәҸпјҢиҰҒд№ҲйғҪжҳҜDESCи§„еҲҷжҺ’еәҸгҖӮ

  4. жҺ’еәҸеҲ—дҪҝвҪӨдәҶеӨҚжқӮзҡ„иЎЁиҫҫејҸ

  • ж•°жҚ®зұ»еһӢи¶Ҡвј©пјҢеңЁжҹҘиҜўж—¶иҝӣвҫҸзҡ„вҪҗиҫғж“ҚдҪңи¶Ҡеҝ«
  • ж•°жҚ®зұ»еһӢи¶Ҡе°ҸпјҢзҙўеј•еҚ з”Ёзҡ„еӯҳеӮЁз©әй—ҙе°ұи¶Ҡе°‘пјҢеңЁдёҖдёӘж•°жҚ®йЎөеҶ…е°ұеҸҜд»Ҙж”ҫдёӢжӣҙеӨҡзҡ„и®°еҪ•пјҢд»ҺиҖҢеҮҸе°‘зЈҒзӣҳI/OеёҰжқҘзҡ„жҖ§иғҪжҚҹиҖ—пјҢд№ҹе°ұж„Ҹе‘ізқҖеҸҜд»ҘжҠҠжӣҙеӨҡзҡ„ж•°жҚ®йЎөзј“ еӯҳеңЁеҶ…еӯҳдёӯпјҢд»ҺиҖҢеҠ еҝ«иҜ»еҶҷж•ҲзҺҮ
